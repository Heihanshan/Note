# DML

## 一、插入

````mysql
-- 1、给指定字段添加数据
insert into 表名 (字段名1,字段名2……) values(值1,值2……);
-- 2、给全部字段添加数据
insert into 表名 values(值1,值2……);
-- 3、批量添加数据
insert into 表名 (字段名1,字段名2……) values(值1,值2……),(值1,值2……),(值1,值2……),……;
insert into 表名 values(值1,值2……),(值1,值2……),(值1,值2……),……;
````

## 二、更新和删除

```mysql
-- 指定更新某个元组的字段数据
update 表名 set 某字段=值(更新的) where 某(其中之一)字段=某值(原有的);
-- 更新某元组的多个字段
update 表名 set 字段1=值1(更新的) 字段2=值2(更新的) 字段3=值3(更新的) where 某字段=某值(原有的);
-- 修改所有元组的某字段的值
update 表名 set 字段=值(更新的);

-- 删除某行元组
delete from 表名 where 字段 =某值;
-- 删除所有元组
delete from 表名
-- 删除标的某个字段的所有值
update 表名 set 字段=null;
```
